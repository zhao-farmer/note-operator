    async function sendMessageStream() {
        const message = document.getElementById('message').value;
        const model = document.getElementById('model').value;
        
        // 创建消息容器
        const msgId = appendMessage('ai', '', false);
        const msgElement = document.getElementById(msgId);
        const contentSpan = msgElement.querySelector('.content') || document.createElement('span');
        contentSpan.className = 'content';
        
        try {
            const response = await fetch('http://localhost:11434/api/generate', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    model: model,
                    prompt: message,
                    stream: true  // 启用流式输出
                })
            });
            
            const reader = response.body.getReader();
            const decoder = new TextDecoder();
            
            while (true) {
                const { done, value } = await reader.read();
                if (done) break;
                
                // 解析每一块数据
                const chunk = decoder.decode(value);
                const lines = chunk.split('\n');
                
                for (const line of lines) {
                    if (line.trim()) {
                        try {
                            const json = JSON.parse(line);
                            contentSpan.textContent += json.response;
                        } catch (e) {
                            console.error('解析错误:', e);
                        }
                    }
                }
            }
            
            msgElement.innerHTML = `<strong>AI:</strong> `;
            msgElement.appendChild(contentSpan);
            
        } catch (error) {
            msgElement.innerHTML = `<strong>AI:</strong> 错误：${error.message}`;
        }
    }

import React, { useState } from 'react';

function OllamaChat() {
    const [messages, setMessages] = useState([]);
    const [input, setInput] = useState('');
    const [loading, setLoading] = useState(false);
    const [model, setModel] = useState('qwen3:1.7b');

    const sendMessage = async () => {
        if (!input.trim()) return;
        
        // 添加用户消息
        const userMsg = { role: 'user', content: input };
        setMessages(prev => [...prev, userMsg]);
        setInput('');
        setLoading(true);
        
        try {
            const response = await fetch('http://localhost:11434/api/chat', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    model: model,
                    messages: [...messages, userMsg].map(m => ({
                        role: m.role,
                        content: m.content
                    })),
                    stream: false
                })
            });
            
            const data = await response.json();
            
            // 添加AI回复
            setMessages(prev => [...prev, {
                role: 'assistant',
                content: data.message.content
            }]);
            
        } catch (error) {
            alert('错误：' + error.message);
        } finally {
            setLoading(false);
        }
    };

    return (
        <div style={{ maxWidth: '800px', margin: '0 auto', padding: '20px' }}>
            <h2>Ollama React 聊天</h2>
            
            <select value={model} onChange={(e) => setModel(e.target.value)}>
                <option value="qwen3:1.7b">Qwen3 1.7B</option>
                <option value="qwen3:7b">Qwen3 7B</option>
            </select>
            
            <div style={{ height: '400px', overflowY: 'auto', border: '1px solid #ccc', padding: '10px', margin: '10px 0' }}>
                {messages.map((msg, idx) => (
                    <div key={idx} style={{ color: msg.role === 'user' ? 'blue' : 'green' }}>
                        <strong>{msg.role === 'user' ? '你' : 'AI'}:</strong> {msg.content}
                    </div>
                ))}
                {loading && <div style={{ color: 'gray' }}>AI: 正在思考...</div>}
            </div>
            
            <div style={{ display: 'flex', gap: '10px' }}>
                <input
                    style={{ flex: 1, padding: '10px' }}
                    value={input}
                    onChange={(e) => setInput(e.target.value)}
                    onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
                    placeholder="输入消息..."
                />
                <button onClick={sendMessage} disabled={loading}>发送</button>
            </div>
        </div>
    );
}

export default OllamaChat;

# main.py
from fastapi import FastAPI, HTTPException, Header
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from typing import Optional, List
import httpx
import asyncio
import json

app = FastAPI()

OLLAMA_HOST = "http://localhost:11434"

class GenerateRequest(BaseModel):
    prompt: str
    model: str = "qwen3:1.7b"
    stream: bool = False
    system: Optional[str] = None
    temperature: Optional[float] = 0.7

class ChatMessage(BaseModel):
    role: str
    content: str

class ChatRequest(BaseModel):
    messages: List[ChatMessage]
    model: str = "qwen3:1.7b"
    stream: bool = False

@app.post("/api/generate")
async def generate(request: GenerateRequest):
    async with httpx.AsyncClient() as client:
        payload = {
            "model": request.model,
            "prompt": request.prompt,
            "stream": request.stream
        }

        if request.system:
            payload["system"] = request.system
        if request.temperature:
            payload["options"] = {"temperature": request.temperature}

        if request.stream:
            async def stream_generator():
                async with client.stream("POST", f"{OLLAMA_HOST}/api/generate",     json=payload) as response:
                    async for line in response.aiter_lines():
                        if line:
                            yield f"data: {line}\n\n"
            return StreamingResponse(stream_generator(), media_type="text/  event-stream")
        else:
            response = await client.post(f"{OLLAMA_HOST}/api/generate", json=payload)
            return response.json()

@app.post("/api/chat")
async def chat(request: ChatRequest):
    async with httpx.AsyncClient() as client:
        payload = {
            "model": request.model,
            "messages": [msg.dict() for msg in request.messages],
            "stream": request.stream
        }

        response = await client.post(f"{OLLAMA_HOST}/api/chat", json=payload)
        return response.json()

@app.get("/api/models")
async def list_models():
    async with httpx.AsyncClient() as client:
        response = await client.get(f"{OLLAMA_HOST}/api/tags")
        return response.json()

// 流式生成
app.post('/api/generate-stream', async (req, res) => {
    const { prompt, model = 'qwen3:1.7b' } = req.body;

    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');

    try {
        const response = await axios({
            method: 'post',
            url: `${OLLAMA_URL}/api/generate`,
            data: { model, prompt, stream: true },
            responseType: 'stream'
        });

        response.data.on('data', chunk => {
            res.write(`data: ${chunk.toString()}\n\n`);
        });

        response.data.on('end', () => {
            res.write('data: [DONE]\n\n');
            res.end();
        });

    } catch (error) {
        res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
        res.end();
    }
});

// OllamaController.java
@RestController
@RequestMapping("/api/ollama")
public class OllamaController {

    private final RestTemplate restTemplate = new RestTemplate();
    private final String OLLAMA_URL = "http://localhost:11434";

    @PostMapping("/generate")
    public ResponseEntity<?> generate(@RequestBody GenerateRequest request) {
        try {
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_JSON);

            Map<String, Object> payload = new HashMap<>();
            payload.put("model", request.getModel());
            payload.put("prompt", request.getPrompt());
            payload.put("stream", false);

            HttpEntity<Map<String, Object>> entity = new HttpEntity<>(payload,  headers);

            ResponseEntity<Map> response = restTemplate.postForEntity(
                OLLAMA_URL + "/api/generate", 
                entity, 
                Map.class
            );

            return ResponseEntity.ok(response.getBody());

        } catch (Exception e) {
            return ResponseEntity.status(500).body(Map.of("error", e.getMessage()));
        }
    }

    @GetMapping("/models")
    public ResponseEntity<?> listModels() {
        try {
            ResponseEntity<Map> response = restTemplate.getForEntity(
                OLLAMA_URL + "/api/tags", 
                Map.class
            );
            return ResponseEntity.ok(response.getBody());
        } catch (Exception e) {
            return ResponseEntity.status(500).body(Map.of("error", e.getMessage()));
        }
    }
}

// main.go
package main

import (
    "bytes"
    "encoding/json"
    "io"
    "net/http"
    "github.com/gin-gonic/gin"
)

type GenerateRequest struct {
    Model  string `json:"model"`
    Prompt string `json:"prompt"`
    Stream bool   `json:"stream"`
}

type ChatRequest struct {
    Model    string        `json:"model"`
    Messages []ChatMessage `json:"messages"`
}

type ChatMessage struct {
    Role    string `json:"role"`
    Content string `json:"content"`
}

func main() {
    r := gin.Default()

    r.POST("/api/generate", handleGenerate)
    r.POST("/api/chat", handleChat)
    r.GET("/api/models", handleListModels)

    r.Run(":8080")
}

func handleGenerate(c *gin.Context) {
    var req GenerateRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }

    // 调用Ollama
    ollamaReq := map[string]interface{}{
        "model":  req.Model,
        "prompt": req.Prompt,
        "stream": req.Stream,
    }

    jsonData, _ := json.Marshal(ollamaReq)

    resp, err := http.Post(
        "http://localhost:11434/api/generate",
        "application/json",
        bytes.NewBuffer(jsonData),
    )

    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }
    defer resp.Body.Close()

    if req.Stream {
        // 流式输出
        c.Stream(func(w io.Writer) bool {
            buf := make([]byte, 1024)
            n, err := resp.Body.Read(buf)
            if err != nil {
                return false
            }
            c.SSEvent("message", string(buf[:n]))
            return true
        })
    } else {
        var result map[string]interface{}
        json.NewDecoder(resp.Body).Decode(&result)
        c.JSON(200, result)
    }
}

func handleChat(c *gin.Context) {
    var req ChatRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }

    // 调用Ollama chat API
    ollamaReq := map[string]interface{}{
        "model":    req.Model,
        "messages": req.Messages,
    }

    jsonData, _ := json.Marshal(ollamaReq)

    resp, err := http.Post(
        "http://localhost:11434/api/chat",
        "application/json",
        bytes.NewBuffer(jsonData),
    )

    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }
    defer resp.Body.Close()

    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    c.JSON(200, result)
}

func handleListModels(c *gin.Context) {
    resp, err := http.Get("http://localhost:11434/api/tags")
    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }
    defer resp.Body.Close()

    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    c.JSON(200, result)
}

# api_gateway/main.py
from fastapi import FastAPI, HTTPException, Request
from fastapi.middleware.cors import CORSMiddleware
import httpx
import asyncio
from typing import Optional
import redis.asyncio as redis
import json
import time
from prometheus_client import Counter, Histogram, generate_latest

app = FastAPI()

# 允许跨域
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

# Redis客户端
redis_client = redis.from_url("redis://redis:6379", decode_responses=True)

# 指标监控
REQUEST_COUNT = Counter('ollama_requests_total', 'Total requests')
REQUEST_TIME = Histogram('ollama_request_duration_seconds', 'Request duration')

# 负载均衡配置
OLLAMA_NODES = [
    "http://ollama-1:11434",
    "http://ollama-2:11434",
    "http://ollama-3:11434"
]
current_node = 0

async def get_next_node():
    """简单的轮询负载均衡"""
    global current_node
    node = OLLAMA_NODES[current_node]
    current_node = (current_node + 1) % len(OLLAMA_NODES)
    return node

@app.post("/v1/chat/completions")
async def chat_completions(request: Request):
    REQUEST_COUNT.inc()
    start_time = time.time()
    
    body = await request.json()
    messages = body.get("messages", [])
    model = body.get("model", "qwen3:1.7b")
    stream = body.get("stream", False)
    
    # 检查缓存
    cache_key = f"chat:{model}:{json.dumps(messages)}"
    cached = await redis_client.get(cache_key)
    
    if cached and not stream:
        REQUEST_TIME.observe(time.time() - start_time)
        return json.loads(cached)
    
    # 选择节点
    node = await get_next_node()
    
    try:
        async with httpx.AsyncClient() as client:
            payload = {
                "model": model,
                "messages": messages,
                "stream": stream
            }
            
            if stream:
                async def stream_generator():
                    async with client.stream("POST", f"{node}/api/chat", json=payload) as response:
                        async for line in response.aiter_lines():
                            if line:
                                yield f"data: {line}\n\n"
                    yield "data: [DONE]\n\n"
                
                return StreamingResponse(stream_generator(), media_type="text/event-stream")
            else:
                response = await client.post(f"{node}/api/chat", json=payload)
                result = response.json()
                
                # 缓存结果（1小时）
                await redis_client.setex(cache_key, 3600, json.dumps(result))
                
                REQUEST_TIME.observe(time.time() - start_time)
                return result
                
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/metrics")
async def metrics():
    return Response(content=generate_latest(), media_type="text/plain")

@app.get("/v1/models")
async def list_models():
    """列出所有可用模型"""
    models = []
    for node in OLLAMA_NODES:
        try:
            async with httpx.AsyncClient() as client:
                response = await client.get(f"{node}/api/tags")
                models.extend(response.json().get("models", []))
        except:
            pass
    
    # 去重
    seen = set()
    unique_models = []
    for model in models:
        if model["name"] not in seen:
            seen.add(model["name"])
            unique_models.append(model)
    
    return {"models": unique_models}

# function_calling.py
import json
import requests

class OllamaFunctionCalling:
    def __init__(self, model="qwen3:7b"):
        self.model = model
        self.functions = {}
    
    def register_function(self, name, description, parameters, func):
        """注册一个可调用的函数"""
        self.functions[name] = {
            "description": description,
            "parameters": parameters,
            "function": func
        }
    
    def call_with_functions(self, prompt):
        """调用模型并可能执行函数"""
        
        # 构建函数描述
        functions_desc = []
        for name, info in self.functions.items():
            functions_desc.append({
                "name": name,
                "description": info["description"],
                "parameters": info["parameters"]
            })
        
        # 调用模型
        response = requests.post(
            "http://localhost:11434/api/generate",
            json={
                "model": self.model,
                "prompt": f"""
                用户问题: {prompt}
                
                可用函数: {json.dumps(functions_desc, ensure_ascii=False)}
                
                如果需要调用函数，请以JSON格式返回：
                {{"function": "函数名", "arguments": {{参数}}}}
                
                如果不需要调用函数，直接回答用户。
                """,
                "stream": False
            }
        )
        
        result = response.json()["response"]
        
        # 解析是否需要调用函数
        try:
            function_call = json.loads(result)
            if "function" in function_call:
                func_name = function_call["function"]
                args = function_call.get("arguments", {})
                
                if func_name in self.functions:
                    # 执行函数
                    func_result = self.functions[func_name]["function"](**args)
                    
                    # 将函数结果返回给模型进行最终回答
                    final_response = requests.post(
                        "http://localhost:11434/api/generate",
                        json={
                            "model": self.model,
                            "prompt": f"""
                            函数 {func_name} 返回结果: {func_result}
                            
                            请根据这个结果回答用户的原始问题: {prompt}
                            """,
                            "stream": False
                        }
                    )
                    return final_response.json()["response"]
        except:
            pass
        
        return result

# rag_system.py
import requests
import numpy as np
from sentence_transformers import SentenceTransformer
import faiss
import pickle

class RAGSystem:
    def __init__(self, model="qwen3:7b"):
        self.model = model
        self.embedder = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')
        self.documents = []
        self.index = None
    
    def add_documents(self, texts):
        """添加文档到知识库"""
        start_idx = len(self.documents)
        self.documents.extend(texts)
        
        # 生成向量
        embeddings = self.embedder.encode(texts)
        
        # 构建FAISS索引
        if self.index is None:
            self.index = faiss.IndexFlatL2(embeddings.shape[1])
        
        self.index.add(np.array(embeddings).astype('float32'))
        
        # 保存索引
        with open('documents.pkl', 'wb') as f:
            pickle.dump(self.documents, f)
        faiss.write_index(self.index, 'index.faiss')
    
    def search(self, query, k=3):
        """搜索相关文档"""
        query_vector = self.embedder.encode([query])
        distances, indices = self.index.search(
            np.array(query_vector).astype('float32'), 
            k
        )
        
        results = []
        for i, idx in enumerate(indices[0]):
            if idx < len(self.documents):
                results.append({
                    'text': self.documents[idx],
                    'score': float(distances[0][i])
                })
        
        return results
    
    def query(self, question, use_rag=True):
        """带RAG的查询"""
        if use_rag and self.index is not None:
            # 检索相关文档
            relevant_docs = self.search(question)
            context = "\n".join([doc['text'] for doc in relevant_docs])
            
            prompt = f"""
            基于以下参考信息回答问题。
            
            参考信息：
            {context}
            
            问题：{question}
            
            请基于参考信息给出准确、简洁的回答。如果参考信息不足以回答问题，请说明。
            """
        else:
            prompt = question
        
        response = requests.post(
            "http://localhost:11434/api/generate",
            json={
                "model": self.model,
                "prompt": prompt,
                "stream": False
            }
        )
        
        return {
            'answer': response.json()['response'],
            'sources': relevant_docs if use_rag and self.index is not None else []
        }

# 使用示例
rag = RAGSystem()

# 添加文档

    "Ollama是一个开源的本地大模型运行框架，支持macOS、Linux和Windows。",
    "Ollama支持Llama 3、DeepSeek、Qwen等多种主流模型。",
    "通过Ollama的API，可以将AI能力集成到各种应用中。"
])

# 查询
result = rag.query("Ollama支持哪些操作系统？")
print(result['answer'])
print("来源:", result['sources'])
